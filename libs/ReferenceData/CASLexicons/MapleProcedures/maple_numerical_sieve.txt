customResultSieve := proc(values::list)
  local result, output, i;
  output := NULL;
  for i from 1 to nops(values) do
    result := values[i][1];
    try
      if not #TEST# then
        output := output,values[i];
      end if;
    catch:
      output := output,[Error, values[i][1]];
    end try;
  end do;
  return [output];
end proc:

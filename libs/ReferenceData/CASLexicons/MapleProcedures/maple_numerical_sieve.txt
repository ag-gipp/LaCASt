customResultSieve := proc(values)
  local result, output, i;
  if not(type(values,list)) then
    return values;
  end if;
  output := NULL;
  for i from 1 to nops(values) do
    result := values[i][1];
    try
      if not #TEST# then
        output := output,values[i];
      end if;
    catch:
      output := output,[Error, values[i][2]];
    end try;
  end do;
  return [output];
end proc:
